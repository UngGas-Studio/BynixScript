Example

This is a simple example of what can be done with BynixScript.

          

            
          

          
var myAge = 12
if myAge.is_type === "number":
  if myAge > 18 || myAge === 18:
    print("Adult")
  else:
    print("Child")
  end
else:
  print("Value must be a number.")
end
